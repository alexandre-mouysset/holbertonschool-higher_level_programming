>>> my_list = MyList()
>>> my_list.append(1)

>>> my_list.append(3)
>>> my_list.append(4)
>>> my_list.append(5)

[1, 2, 3, 4, 5]

>>> my_list = MyList()

>>> my_list.append(2)
>>> my_list.append(8)
>>> my_list.append(1)
>>> my_list.append(9)

[1, 2, 5, 8, 9]

[5, 2, 8, 1, 9]

>>> my_list.append(0)
>>> my_list.append(-5)
>>> my_list.append(3)
